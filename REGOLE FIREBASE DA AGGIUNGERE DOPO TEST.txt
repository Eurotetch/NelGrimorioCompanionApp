// Helper function: check if user is authenticated
    function isAuthenticated() {
      return request.auth != null;
    }
    
    // Helper function: check if user is admin or staff
    function isAdminOrStaff() {
      return isAuthenticated() && 
             get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role in ['admin', 'staff'];
    }
    
    // Users collection
    match /users/{userId} {
      allow read: if isAuthenticated();
      allow create: if isAuthenticated() && request.auth.uid == userId;
      allow update: if isAuthenticated() && 
                       (request.auth.uid == userId || isAdminOrStaff());
      allow delete: if isAdminOrStaff();
    }
    
    // Games collection
    match /games/{gameId} {
      allow read: if true; // Tutti possono vedere i giochi
      allow create, update, delete: if isAdminOrStaff();
    }
    
    // Rooms collection
    match /rooms/{roomId} {
      allow read: if true; // Tutti possono vedere le stanze
      allow create: if isAuthenticated();
      allow update: if isAuthenticated() && 
                       (resource.data.creatorId == request.auth.uid || isAdminOrStaff());
      allow delete: if isAdminOrStaff();
    }
    
    // Personal collections
    match /personalCollections/{userId} {
      allow read: if true; // Tutti possono vedere le collezioni
      allow write: if isAuthenticated() && request.auth.uid == userId;
    }
    
    // Wishlists
    match /wishlists/{userId} {
      allow read: if true;
      allow write: if isAuthenticated() && request.auth.uid == userId;
    }
    
    // Events
    match /events/{eventId} {
      allow read: if true;
      allow create, update, delete: if isAdminOrStaff();
    }
    
    // Settings (global)
    match /settings/{document=**} {
      allow read: if true;
      allow write: if isAdminOrStaff();
    }
